% Frame 4 --- Motivation
\begin{frame}
\frametitle{Motivation}
\begin{columns}
  \column{0.55\textwidth}
  \begin{itemize}
    \item Global data generation exceeds \textbf{2.5 quintillion bytes/day}
    \item Most of this value is \textbf{time-sensitive} --- stale data = lost opportunity
    \item Traditional \textbf{batch pipelines} (Hadoop, scheduled SQL): hourly/daily latency
    \item A music platform cannot recommend based on what a user played \emph{an hour ago}
  \end{itemize}
  \vspace{0.4cm}
  \textbf{Stream processing} enables:
  \begin{itemize}
    \item Real-time personalisation in \textbf{milliseconds}
    \item Operational dashboards reflecting \textbf{current} state
    \item Event-driven microservices \& fault-tolerant pipelines
  \end{itemize}
  \column{0.42\textwidth}
  \centering
  \begin{tikzpicture}
    \node[draw, fill=BKTemplate!15, rounded corners, text width=3.5cm, align=center, minimum height=0.7cm] (batch) {Batch: hours/days latency};
    \node[draw, fill=green!15, rounded corners, text width=3.5cm, align=center, minimum height=0.7cm, below=0.6cm of batch] (stream) {Stream: milliseconds latency};
    \draw[->, thick, red] (batch) -- (stream) node[midway,right,font=\tiny] {paradigm shift};
  \end{tikzpicture}
\end{columns}
\end{frame}

% Frame 5 --- Problem Statement
\begin{frame}
\frametitle{Problem Statement}
A music streaming platform generates \textbf{3 real-time event types}:
\vspace{0.2cm}
\begin{columns}
  \column{0.33\textwidth}
  \centering\textbf{listen\_events}\\
  \small Artist, song, duration, user location, session
  \column{0.33\textwidth}
  \centering\textbf{page\_view\_events}\\
  \small Navigation, HTTP method, status code, user context
  \column{0.33\textwidth}
  \centering\textbf{auth\_events}\\
  \small Login/logout, session validity, subscription level
\end{columns}
\vspace{0.4cm}
\textbf{Why batch is insufficient --- the 4 Vs:}
\begin{itemize}
  \item \textbf{Velocity} --- thousands of events/second; buffering for hours wastes time-sensitivity
  \item \textbf{Volume} --- accumulated data grows rapidly; batch windows grow ever larger
  \item \textbf{Variety} --- 3 schemas, different transforms, must join dimension data (users, songs)
  \item \textbf{Reliability} --- outages must not cause data loss; resume from last committed offset
\end{itemize}
\vspace{0.2cm}
\textbf{Core challenge:} design a fault-tolerant, low-latency pipeline that ingests heterogeneous events, transforms them in real time, and delivers clean data to a cloud warehouse --- with exactly-once guarantees.
\end{frame}

\section{Message Queue: Apache Kafka}
%---------------------------------------------------------

% Frame 8 --- Why Kafka
\begin{frame}
\frametitle{Why Apache Kafka?}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{Problems with direct service communication:}
  \begin{itemize}
    \item \textbf{Tight coupling} --- producers must know every consumer's address \& schema
    \item \textbf{No buffering} --- slow consumer blocks or drops messages
    \item \textbf{No replay} --- message consumed = gone; no recovery on failure
    \item \textbf{Fan-out} --- $N$ consumers = $N$ separate calls per producer
  \end{itemize}
  \column{0.48\textwidth}
  \textbf{How Kafka solves it:}
  \begin{itemize}
    \item \textbf{Decoupling} --- producers write to topics; consumers subscribe independently
    \item \textbf{Durable buffering} --- records persisted to disk \& replicated
    \item \textbf{Replay} --- configurable retention; re-read past events anytime
    \item \textbf{Horizontal scalability} --- partitions linearly scale throughput
    \item \textbf{Fan-out for free} --- multiple consumer groups, each own offset
  \end{itemize}
\end{columns}
\end{frame}

% Frame 9 --- Kafka Architecture
\begin{frame}
\frametitle{Kafka Architecture}
\begin{columns}
  \column{0.52\textwidth}
  \includegraphics[width=\textwidth]{kafka/kafka_architecture.png}
  \column{0.45\textwidth}
  \begin{itemize}
    \item \textbf{Producer} --- partitions by key hash, batches, compresses; \texttt{acks=all} for durability
    \item \textbf{Broker} --- leader/follower replicas; append-only segment log; sequential I/O
    \item \textbf{Topic} --- named channel; immutable records; time-based or log-compaction retention
    \item \textbf{Partition} --- ordered, offset-indexed; unit of parallelism
    \item \textbf{Consumer Group} --- each partition $\rightarrow$ exactly one member; independent offset cursor per group
  \end{itemize}
  \vspace{0.2cm}
  \small\textit{Parallelism rule: max useful consumers = number of partitions}
\end{columns}
\end{frame}

\section{Streaming Processing Engines}

% Frame 12 --- Batch vs Stream
\begin{frame}
\frametitle{Batch vs.\ Stream Processing}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{Batch Processing:}
  \begin{itemize}
    \item Collect data over a window $\rightarrow$ store $\rightarrow$ process in bulk
    \item Latency: \textbf{hours to days}
    \item Tools: Hadoop MapReduce, scheduled SQL
    \item Good for: historical reports, large-scale transforms
  \end{itemize}
  \vspace{0.4cm}
  \textbf{Stream Processing:}
  \begin{itemize}
    \item Computation runs \emph{continuously} as records arrive
    \item Latency: \textbf{milliseconds to seconds}
    \item Tools: Apache Flink, Spark Structured Streaming
    \item Good for: real-time dashboards, fraud detection, IoT
  \end{itemize}
  \column{0.48\textwidth}
  \textbf{Key Challenges in Stream Processing:}
  \begin{enumerate}
    \item \textbf{Unbounded data} --- no defined end; cannot accumulate unbounded state
    \item \textbf{Out-of-order events} --- network delays, clock skew; need watermarks
    \item \textbf{Stateful computation} --- aggregations \& joins require fault-tolerant keyed state
    \item \textbf{Fault tolerance} --- any node can fail; must resume without data loss or duplicates
    \item \textbf{Delivery semantics} --- at-most-once, at-least-once, exactly-once
  \end{enumerate}
\end{columns}
\end{frame}

% Frame 13 --- Apache Spark
\begin{frame}
\frametitle{Apache Spark Structured Streaming}
\begin{columns}
  \column{0.55\textwidth}
  \begin{figure}
    \includegraphics[width=\textwidth]{engine/spark_micro_batch.png}
    \caption*{\tiny Micro-batch: stream treated as sequence of small DataFrames}
  \end{figure}
  \vspace{-0.3cm}
  \begin{figure}
    \includegraphics[width=\textwidth]{engine/spark_rtm.png}
    \caption*{\tiny Continuous Processing (RTM): epoch-based, $\sim$1ms latency}
  \end{figure}
  \column{0.42\textwidth}
  \textbf{Micro-batch (default):}
  \begin{itemize}\small
    \item Trigger interval configurable (e.g.\ 1 s)
    \item WAL + idempotent sinks for fault tolerance
    \item Latency: \textbf{$\sim$100ms--seconds}
    \item Exactly-once with idempotent sinks
  \end{itemize}
  \vspace{0.25cm}
  \textbf{Continuous (RTM):}
  \begin{itemize}\small
    \item Record-at-a-time; epoch checkpointing
    \item Latency: \textbf{$\sim$1ms}
    \item At-least-once semantics
  \end{itemize}
  \vspace{0.25cm}
  \small\textit{Unified API: same code runs in batch and streaming mode}
\end{columns}
\end{frame}

% Frame 14 --- Apache Flink
\begin{frame}
\frametitle{Apache Flink: True Event Streaming}
\begin{columns}
  \column{0.55\textwidth}
  \begin{figure}
    \includegraphics[width=\textwidth]{engine/flink_dataflow_unbounded_stream.png}
    \caption*{\tiny True streaming dataflow: operators process records one-by-one}
  \end{figure}
  \vspace{-0.3cm}
  \begin{figure}
    \includegraphics[width=\textwidth]{engine/flink_checkpoint.png}
    \caption*{\tiny Async barrier snapshotting: barriers flow through the DAG triggering consistent snapshots}
  \end{figure}
  \column{0.42\textwidth}
  \textbf{Dataflow model:}
  \begin{itemize}\small
    \item Records processed \textbf{one-by-one} --- no micro-batching
    \item Operators form a DAG; data flows continuously
    \item \textbf{Event time} + watermarks handle out-of-order events
  \end{itemize}
  \vspace{0.25cm}
  \textbf{Fault tolerance:}
  \begin{itemize}\small
    \item Barrier checkpoints; state in RocksDB
    \item Recovery from last complete checkpoint
    \item \textbf{Exactly-once} with transactional sinks
  \end{itemize}
  \vspace{0.25cm}
  \textbf{APIs:} DataStream, Table API,\\Flink SQL, \textbf{PyFlink}
\end{columns}
\end{frame}

% Frame 15 --- Spark vs Flink comparison
\begin{frame}
\frametitle{Spark vs.\ Flink: Side-by-Side}
\centering
\begin{tabularx}{\textwidth}{lXX}
  \toprule
  \textbf{Dimension}         & \textbf{Spark Structured Streaming} & \textbf{Apache Flink} \\
  \midrule
  Execution model            & Micro-batch (default); Continuous (RTM) & True event-by-event streaming \\
  Latency                    & $\sim$100ms–seconds (micro-batch); $\sim$1ms (RTM) & $\sim$1–10ms \\
  Fault tolerance            & WAL + idempotent sinks                & Async barrier snapshotting \\
  State backend              & In-memory / HDFS                      & RocksDB (incremental) / heap \\
  Exactly-once               & Yes (micro-batch + idempotent sink)   & Yes (transactional sink) \\
  Watermarks                 & Watermark delay column                & Native event-time watermarks \\
  Python support             & PySpark (mature)                      & PyFlink (growing) \\
  Use case sweet spot        & Batch + streaming unified; ML pipelines & Low-latency CEP, stateful pipelines \\
  \bottomrule
\end{tabularx}
\vspace{0.3cm}
\textbf{Why Flink for Streamify:} lower native latency, richer event-time semantics, and \texttt{StatementSet} pattern allows 3 Kafka$\rightarrow$Snowflake pipelines in \textbf{one job} with one shared checkpoint cycle.
\end{frame}

% Frame 18 --- Pipeline
\begin{frame}
\frametitle{Streaming Pipeline: Eventsim $\rightarrow$ Kafka $\rightarrow$ Flink $\rightarrow$ Snowflake}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{Event Simulation (Eventsim):}
  \begin{itemize}\small
    \item 10,000-song subset of Million Songs Dataset
    \item Synthetic user demographics (US census data)
    \item Publishes 3 topics simultaneously with Schema Registry enforcement
  \end{itemize}
  \vspace{0.3cm}
  \textbf{PyFlink Job (\texttt{stream\_events.py}):}
  \begin{itemize}\small
    \item \texttt{create\_kafka\_source()} registers DDL source table per topic
    \item Timestamp: \texttt{TO\_TIMESTAMP\_LTZ(ts/1000, 3)}
    \item Partition columns: \texttt{year\_, month\_, day\_, hour\_}
    \item \texttt{string\_decode} UDF: latin1 $\rightarrow$ UTF-8 for artist/song names
    \item \texttt{StatementSet}: 3 INSERT jobs submitted as \textbf{one Flink job}
  \end{itemize}
  \column{0.48\textwidth}
  \textbf{Kafka Topics:}
  \begin{itemize}\small
    \item \texttt{listen\_events} --- song play metadata
    \item \texttt{page\_view\_events} --- navigation \& HTTP context
    \item \texttt{auth\_events} --- session \& subscription data
  \end{itemize}
  \vspace{0.3cm}
  \textbf{Snowflake Sink:}
  \begin{itemize}\small
    \item JDBC connector with exactly-once guarantees
    \item Raw events land in \texttt{STAGING} schema
    \item \texttt{processed\_time} column tracks pipeline latency
  \end{itemize}
  \vspace{0.3cm}
  \small\textit{One checkpoint cycle for all 3 pipelines --- shared fault recovery}
\end{columns}
\end{frame}

\begin{frame}
\frametitle{Data Modeling \& Orchestration}
\begin{columns}
  \column{0.52\textwidth}
  \textbf{dbt Star Schema (inside Snowflake):}
  \begin{itemize}\small
    \item \texttt{fact\_streams} --- one row per song play; FK to all 5 dims
    \item \texttt{dim\_users} --- demographics \& subscription level
    \item \texttt{dim\_songs} --- title, duration, artist reference
    \item \texttt{dim\_artists} --- name \& location
    \item \texttt{dim\_locations} --- city, state, coordinates
    \item \texttt{dim\_datetime} --- year/month/day/hour/weekday hierarchy
  \end{itemize}
  \vspace{0.2cm}
  \textbf{Wide Table} (\texttt{wide\_streams}):
  \begin{itemize}\small
    \item Pre-joined denormalised view of all 6 tables
    \item Power BI users query \textbf{one flat table} --- no DAX joins needed
  \end{itemize}
  \column{0.45\textwidth}
  \textbf{Apache Airflow Orchestration:}
  \begin{itemize}\small
    \item DAG: \texttt{airflow/dags/dbt.py}
    \item Schedule: \textbf{daily}
    \item Task: \texttt{DockerOperator} runs \texttt{dbt run} in \texttt{dbt-snowflake} container
    \item Docker isolation keeps Airflow worker clean; dbt version pinned
    \item Container auto-removed on success
  \end{itemize}
  \vspace{0.3cm}
  \textbf{ELT Pattern:}\\
  \small Flink \textbf{loads} raw data $\rightarrow$ dbt \textbf{transforms} inside Snowflake.\\
  Streaming concerns cleanly separated from analytical modelling.
\end{columns}
\end{frame}

\section{Conclusion}
%---------------------------------------------------------

% Frame 22 --- Summary
\begin{frame}{Summary: Key Takeaways}
\begin{enumerate}
  \item \textbf{Apache Kafka} provides reliable, replay-capable event transport.
        Producers and consumers are fully decoupled --- the processing engine (Spark or Flink) can be \textbf{swapped without touching the pipeline's upstream or downstream}.

  \vspace{0.3cm}
  \item \textbf{Apache Flink}'s Table API and \texttt{StatementSet} pattern allow 3 independent Kafka$\rightarrow$Snowflake pipelines to run as a \textbf{single job} sharing one checkpoint cycle and one set of TaskManager slots.

  \vspace{0.3cm}
  \item The \textbf{ELT pattern} cleanly separates concerns:
        \begin{itemize}
          \item Flink handles \textbf{real-time loading} of raw events
          \item dbt handles \textbf{analytical transformation} inside Snowflake
          \item Airflow handles \textbf{orchestration and scheduling}
        \end{itemize}

  \vspace{0.3cm}
  \item \textbf{Flink outperforms Spark} for low-latency, stateful streaming workloads --- native event-time semantics, incremental RocksDB state, and millisecond-level latency with exactly-once guarantees.
\end{enumerate}
\end{frame}
