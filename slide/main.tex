\section{Introduction}
%=============================================================================

\begin{frame}{Why Stream Processing?}
\begin{columns}
  \column{0.50\textwidth}
  \textbf{The Problem with Batch}
  \begin{itemize}
    \item \textbf{High latency} — jobs run hourly/daily, insights lag by hours
    \item \textbf{Stale decisions} — by the time data is ready, the moment to act has passed
    \item \textbf{Resource spikes} — large periodic loads strain infra, then sit idle
  \end{itemize}

  \begin{center}
  \begin{tikzpicture}
    \node[draw, fill=red!10, rounded corners, minimum width=3.2cm, minimum height=0.6cm, align=center, font=\small] (b) {Batch: hours of delay};
    \node[draw, fill=green!15, rounded corners, minimum width=3.2cm, minimum height=0.6cm, align=center, font=\small, below=0.5cm of b] (s) {Stream: milliseconds};
    \draw[-{Stealth}, thick, BKTemplate] (b) -- (s);
  \end{tikzpicture}
  \end{center}

  \column{0.47\textwidth}
  \textbf{Stream Processing Enables}
  \begin{itemize}
    \item \textbf{Live monitoring \& alerting} — detect anomalies as they happen
    \item \textbf{Continuous aggregation} — dashboards show current metrics, not a stale snapshot
    \item \textbf{Smooth, steady load} — events processed as they arrive, no periodic resource spikes
  \end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Objectives}
  \textbf{Objectives}
  \begin{enumerate}\setlength\itemsep{6pt}
    \item Overview \textbf{Apache Kafka} as a distributed message queue
    \item Overview \textbf{Apache Flink} as a real-time stream processing engine
    \item Build an end-to-end \textbf{ELT streaming pipeline} — from event simulation to BI dashboard
    \item Demonstrate \textbf{observability} with Prometheus \& Grafana monitoring
  \end{enumerate}
\end{frame}

\section{Message Queue: Apache Kafka}
%=============================================================================

\begin{frame}{Why Apache Kafka?}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{Problems Without a Message Queue}
  \begin{itemize}
    \item \textbf{Tight coupling} — producer knows every consumer
    \item \textbf{No buffering} — slow consumer $\to$ data loss
    \item \textbf{No replay} — consumed = gone forever
    \item \textbf{Fan-out cost} — $N$ consumers = $N$ direct calls
  \end{itemize}

  \column{0.48\textwidth}
  \textbf{How Kafka Solves It}
  \begin{itemize}
    \item \textbf{Decoupling} — write to topic, subscribe independently
    \item \textbf{Durable buffer} — persisted to disk, replicated
    \item \textbf{Replay} — configurable retention, re-read anytime
    \item \textbf{Scalability} — partitions scale throughput linearly
    \item \textbf{Fan-out free} — multiple groups, own offsets
  \end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Kafka Architecture}
\begin{columns}
  \column{0.52\textwidth}
  \includegraphics[width=\textwidth]{kafka/kafka_architecture.png}

  \column{0.45\textwidth}
  \begin{itemize}
    \item \textbf{Producer} — partitions by key; \texttt{acks=all} for durability
    \item \textbf{Broker} — leader/follower replicas; append-only log
    \item \textbf{Topic} — immutable records; retention-based or compacted
    \item \textbf{Partition} — ordered, offset-indexed; unit of parallelism
    \item \textbf{Consumer Group} — one partition $\to$ one member; independent offset per group
  \end{itemize}
  \vspace{0.2cm}
  \small\textit{Max parallelism = number of partitions}
\end{columns}
\end{frame}

\begin{frame}{Kafka Setup in Streamify}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{3-Node KRaft Cluster}
  \begin{itemize}\small
    \item Runs on \textbf{apache/kafka:4.3.0} — no Zookeeper
    \item Each node is both broker \& controller
    \item \textbf{Replication factor 3}, min ISR 2 — no data loss if one broker restarts
  \end{itemize}
  \vspace{0.3cm}
  \textbf{Topic Configuration}
  \begin{itemize}\small
    \item 3 topics: \texttt{listen\_events}, \texttt{page\_view\_events}, \texttt{auth\_events}
    \item \textbf{3 partitions} per topic — enables parallel consumption by Flink
    \item Topics pre-created by \texttt{kafka-init} before producers start
  \end{itemize}

  \column{0.48\textwidth}
  \begin{tikzpicture}[node distance=0.45cm and 0.3cm, every node/.style={font=\tiny}]

\section{Stream Processing: Apache Flink}
%=============================================================================

\begin{frame}{Apache Flink: True Event Streaming}
\begin{columns}
  \column{0.55\textwidth}
  \includegraphics[width=\textwidth]{engine/flink_dataflow_unbounded_stream.png}
  \par\vspace{1pt}{\tiny True streaming: records flow through operator DAG one-by-one}
  \vspace{0.15cm}

  \includegraphics[width=\textwidth]{engine/flink_checkpoint.png}
  \par\vspace{1pt}{\tiny Chandy-Lamport barriers: async snapshots without pausing processing}

  \column{0.42\textwidth}
  \textbf{Dataflow Model}
  \begin{itemize}\small
    \item \textbf{Record-at-a-time} — no batching delay
    \item Operators form a DAG; data flows continuously
    \item \textbf{Event time + watermarks} — correct out-of-order handling
  \end{itemize}
  \vspace{0.25cm}
  \textbf{Fault Tolerance}
  \begin{itemize}\small
    \item Barrier checkpoints injected into the stream
    \item \textbf{Exactly-once} with transactional sinks
    \item State stored in heap or RocksDB
    \item Fully async — no pause in processing
  \end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Flink Setup in Streamify}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{Scaled Deployment}
  \begin{itemize}\small
    \item \textbf{1 JobManager} — coordinates job scheduling and checkpoints
    \item \textbf{3 TaskManagers} — each handles one partition of each topic in parallel
    \item \textbf{Parallelism 3} — matches 3 Kafka partitions exactly
    \item Slot sharing: one full pipeline per TaskManager
  \end{itemize}
  \vspace{0.3cm}
  \textbf{PyFlink Pipeline}
  \begin{itemize}\small
    \item Source: Kafka SQL connector (3 topics)
    \item Transform: Python UDF — JSON decode, timestamp enrichment
    \item Sink: Snowflake via JDBC (exactly-once)
    \item Checkpointing every \textbf{10 seconds} — commits offsets back to Kafka
  \end{itemize}

  \column{0.48\textwidth}
  \includegraphics[width=\textwidth]{engine/flink_architecture.png}
\end{columns}
\end{frame}

\begin{frame}{Monitoring Stack: Prometheus \& Grafana}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{Why Monitor?}
  \begin{itemize}\small
    \item Verify Flink throughtput matches Kafka ingestion rate
    \item Detect consumer-group lag
    \item Track CPU and memory anomalies across all 3 TaskManagers and the JobManager
  \end{itemize}

  \textbf{Components}
  \begin{itemize}\small
    \item \textbf{Flink Prometheus Reporter} — exposes JVM and operator metrics on port 9249
    \item \textbf{kafka-exporter} — exposes consumer-group lag on port 9308
    \item \textbf{Prometheus} — scrapes both; stores time-series data
    \item \textbf{Grafana} — visualises with 3 pre-built dashboards
  \end{itemize}

  \column{0.48\textwidth}
  \textbf{3 Grafana Dashboards}
  \vspace{0.2cm}

  \begin{block}{Kafka Lag}
    \small Consumer-group lag per topic + consumption rate (msgs/s) — detects if Flink is falling behind
  \end{block}

  \begin{block}{Flink Record Processing}
    \small Records ingested per second per source operator — confirms data is flowing through the pipeline
  \end{block}

  \begin{block}{Flink CPU / Memory}
    \small Per-TaskManager JVM CPU load, heap usage, and managed memory — identifies resource pressure
  \end{block}
\end{columns}
\end{frame}

\section{Conclusion}
%=============================================================================

\begin{frame}{Conclusion}
\textbf{Key Takeaways}
\begin{enumerate}\setlength\itemsep{4pt}
  \item \textbf{Kafka} decouples producers from consumers — 3-node KRaft cluster with 3 partitions per topic enables fault tolerance and parallel consumption
  \item \textbf{Flink} delivers sub-10 ms latency with exactly-once guarantees — 3 TaskManagers at parallelism 3 match Kafka partitions for maximum throughput
  \item \textbf{ELT pattern} keeps concerns separate: Flink loads raw events, dbt transforms inside Snowflake through landing $\to$ bronze $\to$ silver $\to$ gold layers
  \item \textbf{Observability} closes the loop — Prometheus \& Grafana confirm the pipeline is healthy in real time
\end{enumerate}

\vspace{0.3cm}
\textbf{Future Directions}
\begin{itemize}
  \item Migrate to cloud — Confluent Cloud + Kinesis + Snowflake Snowpipe
  \item Add schema evolution with Confluent Schema Registry + Avro
\end{itemize}
\end{frame}
