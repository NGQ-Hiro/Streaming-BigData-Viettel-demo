\section{Message Queue: Apache Kafka}
%=============================================================================

% ── Frame 5: Why Kafka? ───────────────────────────────────────────────────────
\begin{frame}{Why Apache Kafka?}
\begin{columns}
  \column{0.48\textwidth}
  \textbf{Problems Without a Message Queue}
  \begin{itemize}
    \item \textbf{Tight coupling} — producer knows every consumer
    \item \textbf{No buffering} — slow consumer → data loss
    \item \textbf{No replay} — consumed = gone forever
    \item \textbf{Fan-out cost} — $N$ consumers = $N$ calls
  \end{itemize}

  \column{0.48\textwidth}
  \textbf{How Kafka Solves It}
  \begin{itemize}
    \item \textbf{Decoupling} — write to topic, subscribe independently
    \item \textbf{Durable buffer} — persisted to disk, replicated
    \item \textbf{Replay} — configurable retention; re-read anytime
    \item \textbf{Scalability} — partitions scale throughput linearly
    \item \textbf{Fan-out free} — multiple groups, own offsets
  \end{itemize}
\end{columns}
\end{frame}

% ── Frame 6: Kafka Architecture ───────────────────────────────────────────────
\begin{frame}{Kafka Architecture}
\begin{columns}
  \column{0.52\textwidth}
  \includegraphics[width=\textwidth]{kafka/kafka_architecture.png}

  \column{0.45\textwidth}
  \begin{itemize}
    \item \textbf{Producer} — partitions by key; batches \& compresses; \texttt{acks=all} for durability
    \item \textbf{Broker} — leader/follower replicas; append-only log; sequential I/O
    \item \textbf{Topic} — immutable records; time-based or log-compaction retention
    \item \textbf{Partition} — ordered, offset-indexed; unit of parallelism
    \item \textbf{Consumer Group} — one partition → one member; independent offset per group
  \end{itemize}
  \vspace{0.2cm}
  \small\textit{Max parallelism = number of partitions}
\end{columns}
\end{frame}

\section{Streaming Processing Engines}
%=============================================================================

% ── Frame 7: Batch vs Stream ──────────────────────────────────────────────────
\begin{frame}{Batch vs.\ Stream Processing}
\begin{columns}
  \column{0.48\textwidth}
  \begin{block}{Batch Processing}
    \begin{itemize}
      \item Collect → store → process in bulk
      \item Latency: \textbf{hours to days}
      \item Tools: Hadoop, Spark, scheduled SQL
    \end{itemize}
  \end{block}
  \vspace{0.3cm}
  \begin{block}{Stream Processing}
    \begin{itemize}
      \item Compute \emph{continuously} as records arrive
      \item Latency: \textbf{milliseconds to seconds}
      \item Tools: Apache Flink, Spark Streaming
    \end{itemize}
  \end{block}

  \column{0.48\textwidth}
  \textbf{Key Challenges in Streaming}
  \begin{enumerate}
    \item \textbf{Unbounded data} — stream never ends; state must be bounded
    \item \textbf{Out-of-order events} — need watermarks to handle late arrivals
    \item \textbf{Fault tolerance} — recover without loss or duplicates; exactly-once semantics
    \item \textbf{State Management} — large state must be checkpointed to durable storage (e.g., local, cloud)
  \end{enumerate}
  \end{columns}
\end{frame}

% ── Frame 8b: DAG-based Dataflow ─────────────────────────────────────────────
\begin{frame}{DAG-based Dataflow: Shared Foundation}
\begin{columns}
  \column{0.55\textwidth}
  \centering
  \includegraphics[width=\textwidth]{engine/dataflow.png}
  \par\vspace{2pt}{\tiny Both Spark and Flink represent computation as a Directed Acyclic Graph (DAG)}

  \column{0.42\textwidth}
  \textbf{What is a Dataflow DAG?}
  \begin{itemize}\small
    \item \textbf{Directed} — data flows one way; no cycles
    \item \textbf{Acyclic} — guarantees deterministic, terminating execution
    \item \textbf{Graph} — \textbf{nodes} are operators (map, filter, join, aggregate); \textbf{edges} are data streams between them
    \item Each operator processes records and passes results downstream
    \item Operators at the same level run \textbf{in parallel} across partitions
    \item Both Spark \& Flink compile your code into a DAG — they differ in \emph{when} and \emph{how} they execute it
  \end{itemize}
\end{columns}
\end{frame}

% ── Frame 9: Apache Flink ────────────────────────────────────────────────────
\begin{frame}{Apache Flink: True Event Streaming}
\begin{columns}
  \column{0.55\textwidth}
  \includegraphics[width=\textwidth]{engine/flink_dataflow_unbounded_stream.png}
  \par\vspace{1pt}{\tiny True streaming: records flow through operator DAG one-by-one}
  \vspace{0.15cm}

  \includegraphics[width=\textwidth]{engine/flink_checkpoint.png}
  \par\vspace{1pt}{\tiny Chandy-Lamport barriers: async snapshots without pausing processing}

  \column{0.42\textwidth}
  \textbf{Dataflow Model}
  \begin{itemize}\small
    \item \textbf{Record-at-a-time} — no batching delay
    \item Operators form a DAG; data flows continuously
    \item \textbf{Event time + watermarks} — correct out-of-order handling
  \end{itemize}
  \vspace{0.25cm}
  \textbf{Fault Tolerance}
  \begin{itemize}\small
    \item Barrier checkpoints injected into streams like a conveyor belt
    \item \textbf{Exactly-once} with transactional sinks
    \item State management can be store in heap or RockDB
    \item Checkpoint is fully async, no pause in processing (local or cloud)
  \end{itemize}
\end{columns}
\end{frame}


% ── Frame 8: Apache Spark ────────────────────────────────────────────────────
\begin{frame}{Apache Spark Structured Streaming}
\begin{columns}
  \column{0.55\textwidth}
  \includegraphics[width=\textwidth]{engine/spark_micro_batch_latency.png}
  % \par\vspace{1pt}{\tiny Micro-batch: stream as a sequence of small DataFrames}
  % \vspace{0.15cm}

  \includegraphics[width=\textwidth]{engine/spark_rtm.png}
  % \par\vspace{1pt}{\tiny Real-time Mode (Spark 4): concurrent stages, streaming shuffle}

  \column{0.42\textwidth}
  \textbf{Micro-batch (default)}
  \begin{itemize}\small
    \item Latency: \textbf{100 ms – seconds}
    \item Still depend on exist batch mode stage, plan create a lot of latency
    \item State management In-memory+WAL or RocksDB
    \item Checkpoint sync write to checkpoint location (local or cloud)
  \end{itemize}
  \vspace{0.25cm}
  \textbf{Real-time Mode (Spark 4)}
  \begin{itemize}\small
    \item Inherit from micro-batch, but update \textbf{concurrent stages} and \textbf{longer duration micro-batches}
    \item Latency: \textbf{sub-100 ms}
  \end{itemize}
  \vspace{0.2cm}
\end{columns}
\end{frame}


% ── Frame 10: Spark vs Flink ─────────────────────────────────────────────────
\begin{frame}{Spark vs.\ Flink: Side-by-Side}
\centering
\begin{tabularx}{\textwidth}{lXX}
  \toprule
  \textbf{Dimension}  & \textbf{Apache Spark}                            & \textbf{Apache Flink} \\
  \midrule
  Execution model     & Micro-batch (scheduled triggers); Real-time Mode (Spark 4) & True record-at-a-time streaming \\
  Latency             & 100 ms–s (micro-batch); sub-100 ms (RTM)         & 1–10 ms \\
  State management    & In-memory / RocksDB (Spark 3.2+); managed per trigger & RocksDB or heap; always-live, keyed state \\
  Checkpoint          & WAL per micro-batch; idempotent sink for EO      & Async Chandy-Lamport barriers; incremental snapshots \\
  Use case            & Unified batch + streaming; ML pipelines          & Low-latency, stateful, CEP, event-time joins \\
  \bottomrule
\end{tabularx}
\vspace{0.3cm}
\end{frame}

\section{Conclusion}
%=============================================================================

% ── Frame 14: Conclusion ──────────────────────────────────────────────────────
\begin{frame}{Conclusion}
\textbf{Key Takeaways}
\begin{enumerate}
  \item \textbf{Kafka} decouples producers from consumers — swap the processing engine without touching the rest of the pipeline
  \item \textbf{Flink} delivers sub-10 ms latency, native event-time semantics, and exactly-once guarantees — outperforming Spark for stateful real-time workloads
  \item \textbf{ELT pattern} keeps concerns separate: Flink loads raw events, dbt transforms inside Snowflake, Airflow orchestrates
\end{enumerate}
\vspace{0.4cm}
